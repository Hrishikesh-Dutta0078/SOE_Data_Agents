# RTB DAX-to-SQL KPI Glossary Translation — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Translate 307 DAX measures from `RTB Dataverse Technical Document 3.xlsx` into executable T-SQL and output them as `server/context/knowledge/kpi-glossary-rtb.json`.

**Architecture:** A Node.js script (`scripts/translateDaxToSql.js`) orchestrates a 6-stage pipeline: Excel parsing → lookup map building → dependency graph → DAX-to-SQL translation → metadata generation → JSON output. The translator uses recursive pattern matching against known DAX constructs (CALCULATE, DIVIDE, SWITCH, VAR/RETURN, etc.) and inlines referenced measures as scalar subqueries.

**Tech Stack:** Node.js 18+ (CommonJS), `xlsx` package (globally installed), `node:test` for testing.

**Spec:** `docs/superpowers/specs/2026-03-31-rtb-dax-to-sql-kpi-glossary-design.md`

```js
// scripts/dax-to-sql/excelReader.js
'use strict';

const XLSX = require('xlsx');
const { TABLE_MAP } = require('./constants');

/**
 * Reads the RTB Dataverse Technical Document Excel and returns
 * structured data: measures, table/column/join maps.
 */
function readExcel(filePath) {
  const wb = XLSX.readFile(filePath);

  const measures = readMeasures(wb);
  const tableMap = buildTableMap(wb);
  const columnsByTable = buildColumnMap(wb);
  const joinMap = buildJoinMap(wb);

  return { measures, tableMap, columnsByTable, joinMap };
}

function readMeasures(wb) {
  const ws = wb.Sheets['Measures'];
  const raw = XLSX.utils.sheet_to_json(ws, { defval: '' });
  // Row 0 is the header row (Measure Name, Table Name, ...)
  return raw.slice(1).map(r => ({
    name: r['EVALUATE INFO.VIEW.MEASURES()'] || '',
    tableName: r.__EMPTY || '',
    dataType: r.__EMPTY_1 || '',
    expression: (r['EVALUATE INFO.VIEW.MEASURES()_1'] || '').replace(/\r\r\n/g, '\n').replace(/\r\n/g, '\n').replace(/\r/g, '\n'),
    isHidden: r.__EMPTY_2 === true,
    displayFolder: r.__EMPTY_3 || '',
    formatString: r.__EMPTY_4 || '',
  }));
}

function buildTableMap(wb) {
  // Start with the hardcoded TABLE_MAP from constants (authoritative).
  // Supplement with any additional entries from the Table Source sheet.
  const ws = wb.Sheets['Table Source'];
  const raw = XLSX.utils.sheet_to_json(ws, { defval: '' });
  const map = { ...TABLE_MAP };

  raw.forEach(r => {
    const pbix = r['[Table/Folder Name]'];
    const src = r['Source Table'];
    if (!pbix || !src) return;
    if (map[pbix] !== undefined) return; // constants.js takes precedence
    if (src === 'Measure Table' || src.includes('Manual Table') || src.includes('DAX Table') || src.includes('Calculated')) {
      map[pbix] = null;
      return;
    }
    const fyMatch = src.match(/^FY Calendar\((.+)\)$/);
    map[pbix] = fyMatch ? fyMatch[1] : src;
  });

  return map;
}

function buildColumnMap(wb) {
  const ws = wb.Sheets['Columns'];
  const raw = XLSX.utils.sheet_to_json(ws, { defval: '' });
  const map = {}; // tableName → Set<columnName>
  raw.slice(1).forEach(r => {
    const table = r.__EMPTY || '';
    const col = r['EVALUATE INFO.VIEW.COLUMNS()'] || '';
    if (!table || !col) return;
    if (!map[table]) map[table] = new Set();
    map[table].add(col);
  });
  return map;
}

function buildJoinMap(wb) {
  const ws = wb.Sheets['Data Model Design'];
  const raw = XLSX.utils.sheet_to_json(ws, { defval: '' });
  const map = {}; // fromTable → { toTable → { fromCol, toCol } }

  raw.slice(1).forEach(r => {
    const from = r.__EMPTY_5 || '';
    const fromCol = r.__EMPTY_6 || '';
    const to = r.__EMPTY_8 || '';
    const toCol = r.__EMPTY_9 || '';
    if (!from || !to || !fromCol || !toCol) return;

    // Forward direction
    if (!map[from]) map[from] = {};
    if (!map[from][to]) map[from][to] = { fromCol, toCol };

    // Reverse direction
    if (!map[to]) map[to] = {};
    if (!map[to][from]) map[to][from] = { fromCol: toCol, toCol: fromCol };
  });

  return map;
}

module.exports = { readExcel, readMeasures, buildTableMap, buildColumnMap, buildJoinMap };
```

```js
// scripts/dax-to-sql/dependencyGraph.js
'use strict';

/**
 * Find measure references in a DAX expression.
 * Returns a Set of measure names that are referenced (not table-qualified columns).
 */
function findMeasureRefs(expression, measureNames) {
  const refs = new Set();
  // Match [Name] — but skip if preceded by a table qualifier like 'TableName' or TableName
  const regex = /\[([^\]]+)\]/g;
  let match;
  while ((match = regex.exec(expression)) !== null) {
    const name = match[1];
    if (!measureNames.has(name)) continue;
    // Check preceding context: if the char before [ is ' or a word char, it's a column ref
    const before = expression.substring(Math.max(0, match.index - 1), match.index);
    if (before === "'" || /\w/.test(before)) continue;
    refs.add(name);
  }
  return refs;
}

/**
 * Build a dependency graph: measureName → Set of measure names it depends on.
 */
function buildDependencyGraph(measures) {
  const measureNames = new Set(measures.map(m => m.name));
  const graph = {};

  measures.forEach(m => {
    graph[m.name] = findMeasureRefs(m.expression || '', measureNames);
  });

  return graph;
}

/**
 * Topological sort using DFS. Returns { order: string[], cycles: string[] }.
 * order[0] is the first to process (no deps), order[last] is the most dependent.
 */
function topologicalSort(graph) {
  const visited = new Set();
  const inStack = new Set();
  const order = [];
  const cycles = [];

  function visit(node) {
    if (inStack.has(node)) {
      cycles.push(node);
      return;
    }
    if (visited.has(node)) return;

    inStack.add(node);
    visited.add(node);

    const deps = graph[node] || new Set();
    for (const dep of deps) {
      if (graph[dep] !== undefined) visit(dep);
    }

    inStack.delete(node);
    order.push(node);
  }

  Object.keys(graph).forEach(node => {
    if (!visited.has(node)) visit(node);
  });

  return { order, cycles };
}

module.exports = { findMeasureRefs, buildDependencyGraph, topologicalSort };
```

```js
// scripts/dax-to-sql/daxParser.js
'use strict';

/**
 * Remove DAX line comments (--- or -- to end of line).
 */
function stripDaxComments(expr) {
  return expr.replace(/---[^\n]*/g, '').replace(/--[^\n]*/g, '');
}

/**
 * Extract a function call: returns { name, innerRaw } or null.
 * innerRaw is the content inside the outermost parentheses.
 */
function extractFunctionCall(expr) {
  const trimmed = expr.trim();
  const funcMatch = trimmed.match(/^([A-Z_][A-Z0-9_.]*)\s*\(/i);
  if (!funcMatch) return null;

  const name = funcMatch[1].toUpperCase();
  const startParen = trimmed.indexOf('(', funcMatch[1].length);
  let depth = 0;
  let endParen = -1;

  for (let i = startParen; i < trimmed.length; i++) {
    if (trimmed[i] === '(' || trimmed[i] === '{') depth++;
    else if (trimmed[i] === ')' || trimmed[i] === '}') {
      depth--;
      if (depth === 0) { endParen = i; break; }
    }
  }

  if (endParen === -1) return null;
  const innerRaw = trimmed.substring(startParen + 1, endParen);
  return { name, innerRaw, rest: trimmed.substring(endParen + 1).trim() };
}

/**
 * Split a string on commas at the top level (depth-0 relative to parens/braces).
 */
function splitTopLevelArgs(str) {
  const args = [];
  let depth = 0;
  let current = '';

  for (let i = 0; i < str.length; i++) {
    const ch = str[i];
    if (ch === '(' || ch === '{') depth++;
    else if (ch === ')' || ch === '}') depth--;

    if (ch === ',' && depth === 0) {
      args.push(current.trim());
      current = '';
    } else {
      current += ch;
    }
  }

  if (current.trim()) args.push(current.trim());
  return args;
}

/**
 * Parse a table[column] reference.
 * Returns { table, column } or null if it's a standalone [MeasureRef].
 */
function parseTableColumnRef(expr) {
  const trimmed = expr.trim();
  // 'Table Name'[Column]
  const quoted = trimmed.match(/^'([^']+)'\[([^\]]+)\]$/);
  if (quoted) return { table: quoted[1], column: quoted[2] };
  // TableName[Column]
  const unquoted = trimmed.match(/^([A-Za-z_]\w*(?:\s+\w+)*)\[([^\]]+)\]$/);
  if (unquoted) return { table: unquoted[1], column: unquoted[2] };
  return null;
}

/**
 * Parse VAR ... RETURN ... blocks.
 * Returns { vars: [{ name, expr }], returnExpr }.
 */
function parseVarReturn(expr) {
  const trimmed = expr.trim();
  const vars = [];
  let remaining = trimmed;

  // Extract VAR bindings — each VAR starts at line beginning or after another VAR block
  const varRegex = /\bVAR\s+(\w+)\s*=\s*/gi;
  let match;
  const varPositions = [];
  while ((match = varRegex.exec(remaining)) !== null) {
    varPositions.push({ name: match[1], start: match.index, exprStart: match.index + match[0].length });
  }

  // Find RETURN position
  const returnMatch = remaining.match(/\bRETURN\s+/i);
  const returnIdx = returnMatch ? remaining.indexOf(returnMatch[0]) : -1;

  for (let i = 0; i < varPositions.length; i++) {
    const vp = varPositions[i];
    const end = (i + 1 < varPositions.length)
      ? varPositions[i + 1].start
      : (returnIdx >= 0 ? returnIdx : remaining.length);
    const varExpr = remaining.substring(vp.exprStart, end).trim();
    vars.push({ name: vp.name, expr: varExpr });
  }

  const returnExpr = returnIdx >= 0
    ? remaining.substring(returnIdx + returnMatch[0].length).trim()
    : '';

  return { vars, returnExpr };
}

/**
 * Identify the top-level pattern of a DAX expression.
 * Returns { type, ...parsed data }.
 */
function parseDaxExpression(rawExpr) {
  const expr = stripDaxComments(rawExpr).trim();

  // Literal number
  if (/^-?\d+(\.\d+)?$/.test(expr)) {
    return { type: 'LITERAL', value: expr };
  }

  // Literal string
  if (/^"[^"]*"$/.test(expr)) {
    return { type: 'LITERAL', value: expr };
  }

  // Standalone measure ref: [MeasureName]
  if (/^\[[^\]]+\]$/.test(expr) && !parseTableColumnRef(expr)) {
    return { type: 'MEASURE_REF', name: expr.slice(1, -1) };
  }

  // Table[Column] ref
  const tcRef = parseTableColumnRef(expr);
  if (tcRef) {
    return { type: 'TABLE_COLUMN_REF', ...tcRef };
  }

  // VAR ... RETURN ...
  if (/^\s*VAR\s+/i.test(expr)) {
    const parsed = parseVarReturn(expr);
    return { type: 'VAR_RETURN', ...parsed };
  }

  // Function call
  const funcCall = extractFunctionCall(expr);
  if (funcCall) {
    const { name, innerRaw, rest } = funcCall;
    const args = splitTopLevelArgs(innerRaw);

    // Map known DAX functions to types
    const funcTypeMap = {
      'CALCULATE': 'CALCULATE',
      'DIVIDE': 'DIVIDE',
      'SUM': 'SUM',
      'SUMX': 'SUMX',
      'SWITCH': 'SWITCH',
      'COUNTROWS': 'COUNTROWS',
      'FILTER': 'FILTER',
      'DISTINCTCOUNT': 'DISTINCTCOUNT',
      'MAX': 'MAX',
      'MIN': 'MIN',
      'MAXX': 'MAXX',
      'ROUND': 'ROUND',
      'RANKX': 'RANKX',
      'TOPN': 'TOPN',
      'IF': 'IF',
      'NOT': 'NOT',
      'ISBLANK': 'ISBLANK',
      'BLANK': 'BLANK',
      'COALESCE': 'COALESCE',
      'SELECTEDVALUE': 'SELECTEDVALUE',
      'ALL': 'ALL',
      'ALLSELECTED': 'ALLSELECTED',
      'VALUES': 'VALUES',
      'HASONEVALUE': 'HASONEVALUE',
      'ABS': 'ABS',
      'FORMAT': 'FORMAT',
      'CONCATENATEX': 'CONCATENATEX',
      'RELATED': 'RELATED',
      'AVERAGEX': 'AVERAGEX',
      'AVERAGE': 'AVERAGE',
      'COUNT': 'COUNT',
      'ISINSCOPE': 'ISINSCOPE',
      'SUMMARIZE': 'SUMMARIZE',
      'ADDCOLUMNS': 'ADDCOLUMNS',
    };

    const type = funcTypeMap[name] || 'FUNCTION';
    return { type, name, args, innerRaw, rest };
  }

  // Binary expression: check for operators at top level
  if (/[+\-*/]/.test(expr) || /[<>=!]/.test(expr) || /\b(AND|OR|&&|\|\|)\b/i.test(expr)) {
    return { type: 'EXPRESSION', raw: expr };
  }

  // Fallback
  return { type: 'RAW', raw: expr };
}

module.exports = {
  stripDaxComments,
  extractFunctionCall,
  splitTopLevelArgs,
  parseTableColumnRef,
  parseDaxExpression,
  parseVarReturn,
};
```

```js
  stripDaxComments,
  extractFunctionCall,
  splitTopLevelArgs,
  parseTableColumnRef,
  parseDaxExpression,
  parseVarReturn,

const { TABLE_MAP, TABLE_ALIAS } = require('./constants');

/**
 * Create translation context shared across all measure translations.
 */
function createTranslationContext({ tableMap, tableAlias, joinMap, resolvedMeasures }) {
  return {
    tableMap: tableMap || TABLE_MAP,
    tableAlias: tableAlias || TABLE_ALIAS,
    joinMap: joinMap || {},
    resolvedMeasures: resolvedMeasures || {},
    warnings: [],
    relatedTables: new Set(),
    relatedColumns: new Set(),
    confidence: 'mapped',
  };
}

/**
 * Resolve a PBIX table name to its SQL view name.
 */
function resolveTable(pbixTable, ctx) {
  const view = ctx.tableMap[pbixTable];
  if (view === null) {
    ctx.warnings.push(`Table "${pbixTable}" has no SQL equivalent (manual/DAX table)`);
    ctx.confidence = downgrade(ctx.confidence, 'pbix_only');
    return pbixTable; // fallback
  }
  if (view === undefined) {
    ctx.warnings.push(`Table "${pbixTable}" not found in table map`);
    ctx.confidence = downgrade(ctx.confidence, 'inferred');
    return pbixTable;
  }
  return view;
}

/**
 * Get a short alias for a SQL view.
 */
function getAlias(view, ctx) {
  if (ctx.tableAlias[view]) return ctx.tableAlias[view];
  // Generate from initials
  return view.replace(/^vw_[A-Z]*_EBI_/i, '').substring(0, 3).toLowerCase();
}

function downgrade(current, to) {
  const order = { 'mapped': 0, 'inferred': 1, 'pbix_only': 2 };
  return order[to] > order[current] ? to : current;
}

/**
 * Translate a DAX expression to T-SQL.
 * Returns { sql: string, confidence: string, relatedTables: string[], relatedColumns: string[], warnings: string[] }

  // Reset per-measure tracking
  ctx.relatedTables = new Set();
  ctx.relatedColumns = new Set();
  ctx.confidence = 'mapped';
  ctx.warnings = [];

  const sql = translateExpr(rawExpr, ctx);

  return {
    sql,
    confidence: ctx.confidence,
    relatedTables: [...ctx.relatedTables],
    relatedColumns: [...ctx.relatedColumns],
    warnings: [...ctx.warnings],
  };
}

    case 'SUMMARIZE':
    case 'ADDCOLUMNS':
      ctx.confidence = downgrade(ctx.confidence, 'inferred');
      return `/* ${parsed.name}(...) — complex DAX, manual review needed */`;

    case 'EXPRESSION':
      return translateBinaryExpr(parsed.raw, ctx);

    case 'FUNCTION':
      ctx.confidence = downgrade(ctx.confidence, 'inferred');
      return `/* Unhandled: ${parsed.name}(...) */`;

    case 'RAW':
    default:
      return translateBinaryExpr(parsed.raw || expr, ctx);
  }
}

function translateCalculate(parsed, ctx) {
  const args = parsed.args;
  if (args.length === 0) return 'NULL';

  const measureExpr = args[0];
  const filterExprs = args.slice(1);

  // Translate the measure expression
  const measureSql = translateExpr(measureExpr, ctx);

  // If measureSql is a scalar subquery (SELECT ...), inject WHERE clauses
  // Otherwise wrap it
  const filterClauses = [];
  const joinClauses = [];

  filterExprs.forEach(f => {
    const filterResult = translateFilterArg(f.trim(), ctx);
    if (filterResult.where) filterClauses.push(filterResult.where);
    if (filterResult.join) joinClauses.push(filterResult.join);
  });

  if (filterClauses.length === 0 && joinClauses.length === 0) return measureSql;

  // Try to inject filters into existing subquery
  const subqueryMatch = measureSql.match(/^\(SELECT (.+?) FROM (.+?)\)$/is);
  if (subqueryMatch) {
    const selectPart = subqueryMatch[1];
    let fromPart = subqueryMatch[2];
    const joinStr = joinClauses.length > 0 ? ' ' + joinClauses.join(' ') : '';
    const whereStr = filterClauses.length > 0 ? ' WHERE ' + filterClauses.join(' AND ') : '';

    // Check if FROM already has WHERE
    const whereIdx = fromPart.toUpperCase().indexOf(' WHERE ');
    if (whereIdx >= 0 && filterClauses.length > 0) {
      return `(SELECT ${selectPart} FROM ${fromPart}${joinStr} AND ${filterClauses.join(' AND ')})`;
    }
    return `(SELECT ${selectPart} FROM ${fromPart}${joinStr}${whereStr})`;
  }

  // Wrap as subquery with filters
  const joinStr = joinClauses.join(' ');
  const whereStr = filterClauses.length > 0 ? ' WHERE ' + filterClauses.join(' AND ') : '';
  return `(SELECT ${measureSql}${joinStr ? ' ' + joinStr : ''}${whereStr})`;
}

/**
 * Translate a CALCULATE filter argument to { where, join }.
 */
function translateFilterArg(filterExpr, ctx) {
  const trimmed = filterExpr.trim();

  // ALL('Table') — removes filters, ignore
  if (/^ALL\s*\(/i.test(trimmed) || /^ALLSELECTED\s*\(/i.test(trimmed)) {
    return {};
  }

  // NOT 'Table'[Col] IN {...}
  const notInMatch = trimmed.match(/^NOT\s+'([^']+)'\[([^\]]+)\]\s+IN\s+\{(.+)\}$/is);
  if (notInMatch) {
    const view = resolveTable(notInMatch[1], ctx);
    const alias = getAlias(view, ctx);
    const values = notInMatch[3].split(',').map(v => v.trim().replace(/"/g, "'"));
    ctx.relatedTables.add(view);
    ctx.relatedColumns.add(`${view}.${notInMatch[2]}`);
    return { where: `${alias}.${notInMatch[2]} NOT IN (${values.join(', ')})` };
  }

  // 'Table'[Col] IN {"val1", "val2"}
  const inMatch = trimmed.match(/^'([^']+)'\[([^\]]+)\]\s+IN\s+\{(.+)\}$/is);
  if (inMatch) {
    const view = resolveTable(inMatch[1], ctx);
    const alias = getAlias(view, ctx);
    const values = inMatch[3].split(',').map(v => v.trim().replace(/"/g, "'"));
    ctx.relatedTables.add(view);
    ctx.relatedColumns.add(`${view}.${inMatch[2]}`);
    return { where: `${alias}.${inMatch[2]} IN (${values.join(', ')})` };
  }

  // Unquoted Table[Col] IN {values}
  const inMatchUnquoted = trimmed.match(/^([A-Za-z_][\w\s]*)\[([^\]]+)\]\s+IN\s+\{(.+)\}$/is);
  if (inMatchUnquoted) {
    const view = resolveTable(inMatchUnquoted[1].trim(), ctx);
    const alias = getAlias(view, ctx);
    const values = inMatchUnquoted[3].split(',').map(v => v.trim().replace(/"/g, "'"));
    ctx.relatedTables.add(view);
    ctx.relatedColumns.add(`${view}.${inMatchUnquoted[2]}`);
    return { where: `${alias}.${inMatchUnquoted[2]} IN (${values.join(', ')})` };
  }

  // 'Table'[Col] = value or 'Table'[Col] <> value etc.
  const eqMatch = trimmed.match(/^'([^']+)'\[([^\]]+)\]\s*(=|<>|!=|<=|>=|<|>)\s*(.+)$/is);
  if (eqMatch) {
    const view = resolveTable(eqMatch[1], ctx);
    const alias = getAlias(view, ctx);
    let value = eqMatch[4].trim().replace(/"/g, "'");
    if (/^TRUE\(\)$/i.test(value)) value = '1';
    if (/^FALSE\(\)$/i.test(value)) value = '0';
    ctx.relatedTables.add(view);
    ctx.relatedColumns.add(`${view}.${eqMatch[2]}`);
    return { where: `${alias}.${eqMatch[2]} ${eqMatch[3]} ${value}` };
  }

  // Unquoted Table[Col] = value
  const eqMatchUnquoted = trimmed.match(/^([A-Za-z_][\w\s]*)\[([^\]]+)\]\s*(=|<>|!=|<=|>=|<|>)\s*(.+)$/is);
  if (eqMatchUnquoted) {
    const view = resolveTable(eqMatchUnquoted[1].trim(), ctx);
    const alias = getAlias(view, ctx);
    let value = eqMatchUnquoted[4].trim().replace(/"/g, "'");
    ctx.relatedTables.add(view);
    ctx.relatedColumns.add(`${view}.${eqMatchUnquoted[2]}`);
    return { where: `${alias}.${eqMatchUnquoted[2]} ${eqMatchUnquoted[3]} ${value}` };
  }

  // FILTER() as a filter arg — translate inline
  if (/^FILTER\s*\(/i.test(trimmed)) {
    const filterSql = translateExpr(trimmed, ctx);
    return { where: `/* FILTER: ${filterSql} */` };
  }

  // Fallback
  ctx.confidence = downgrade(ctx.confidence, 'inferred');
  return { where: `/* filter: ${trimmed} */` };
}

function translateDivide(parsed, ctx) {
  const numerator = translateExpr(parsed.args[0], ctx);
  const denominator = translateExpr(parsed.args[1], ctx);
  const altResult = parsed.args.length > 2 ? translateExpr(parsed.args[2], ctx) : null;

  if (altResult) {
    return `CASE WHEN (${denominator}) = 0 OR (${denominator}) IS NULL THEN ${altResult} ELSE (${numerator}) / (${denominator}) END`;
  }
  return `(${numerator}) / NULLIF((${denominator}), 0)`;
}

function translateSwitch(parsed, ctx) {
  const args = parsed.args;

  // SWITCH(TRUE(), cond1, val1, cond2, val2, ..., default)
  if (args[0].trim().toUpperCase() === 'TRUE()') {
    const pairs = args.slice(1);
    let sql = 'CASE';
    for (let i = 0; i < pairs.length - 1; i += 2) {
      const cond = translateExpr(pairs[i], ctx);
      const val = translateExpr(pairs[i + 1], ctx);
      sql += ` WHEN ${cond} THEN ${val}`;
    }
    // If odd number of remaining args, last is default
    if (pairs.length % 2 === 1) {
      sql += ` ELSE ${translateExpr(pairs[pairs.length - 1], ctx)}`;
    }
    sql += ' END';
    return sql;
  }

  // SWITCH(expr, val1, result1, val2, result2, ..., default)
  const switchExpr = translateExpr(args[0], ctx);
  let sql = `CASE ${switchExpr}`;
  const rest = args.slice(1);
  for (let i = 0; i < rest.length - 1; i += 2) {
    sql += ` WHEN ${translateExpr(rest[i], ctx)} THEN ${translateExpr(rest[i + 1], ctx)}`;
  }
  if (rest.length % 2 === 1) {
    sql += ` ELSE ${translateExpr(rest[rest.length - 1], ctx)}`;
  }
  sql += ' END';
  return sql;
}

function translateIf(parsed, ctx) {
  const cond = translateExpr(parsed.args[0], ctx);
  const thenVal = translateExpr(parsed.args[1], ctx);
  const elseVal = parsed.args.length > 2 ? translateExpr(parsed.args[2], ctx) : 'NULL';
  return `CASE WHEN ${cond} THEN ${thenVal} ELSE ${elseVal} END`;
}

function translateVarReturn(parsed, ctx) {
  // Translate each VAR as a CTE-like inline, then compose RETURN
  const varValues = {};

  for (const v of parsed.vars) {
    // Replace previous VAR references in this var's expression
    let expr = v.expr;
    for (const [prevName, prevSql] of Object.entries(varValues)) {
      // Replace standalone occurrences of the var name
      const varRegex = new RegExp(`\\b${escapeRegex(prevName)}\\b`, 'g');
      expr = expr.replace(varRegex, `(${prevSql})`);
    }
    const sql = translateExpr(expr, ctx);
    varValues[v.name] = sql;
  }

  // Translate RETURN expression, substituting VARs
  let returnExpr = parsed.returnExpr;
  for (const [name, sql] of Object.entries(varValues)) {
    const varRegex = new RegExp(`\\b${escapeRegex(name)}\\b`, 'g');
    returnExpr = returnExpr.replace(varRegex, `(${sql})`);
  }

  return translateExpr(returnExpr, ctx);
}

/**
 * Translate a binary expression (operators at top level).
 * Handles: +, -, *, /, comparisons, &&, ||, AND, OR.
 */
function translateBinaryExpr(expr, ctx) {
  const trimmed = expr.trim();

  // Replace DAX-specific tokens
  let sql = trimmed;

  // Replace && with AND, || with OR
  sql = sql.replace(/&&/g, ' AND ');
  sql = sql.replace(/\|\|/g, ' OR ');

  // Replace TRUE() / FALSE()
  sql = sql.replace(/\bTRUE\(\)/gi, '1');
  sql = sql.replace(/\bFALSE\(\)/gi, '0');

  // Replace BLANK() with NULL
  sql = sql.replace(/\bBLANK\(\)/gi, 'NULL');

  // Translate table[column] references and measure refs inline
  // 'Table'[Column] refs
  sql = sql.replace(/'([^']+)'\[([^\]]+)\]/g, (match, table, column) => {
    const view = resolveTable(table, ctx);
    const alias = getAlias(view, ctx);
    ctx.relatedTables.add(view);
    ctx.relatedColumns.add(`${view}.${column}`);
    return `${alias}.${column}`;
  });

  // Unquoted Table[Column] refs (but not [MeasureRef])
  sql = sql.replace(/([A-Za-z_]\w*)\[([^\]]+)\]/g, (match, table, column) => {
    const view = resolveTable(table, ctx);
    if (view) {
      const alias = getAlias(view, ctx);
      ctx.relatedTables.add(view);
      ctx.relatedColumns.add(`${view}.${column}`);
      return `${alias}.${column}`;
    }
    return match;
  });

  // Standalone [MeasureRef]
  sql = sql.replace(/\[([^\]]+)\]/g, (match, name) => {
    const resolved = ctx.resolvedMeasures[name];
    if (resolved) {
      resolved.relatedTables.forEach(t => ctx.relatedTables.add(t));
      resolved.relatedColumns.forEach(c => ctx.relatedColumns.add(c));
      return resolved.sql;
    }
    ctx.confidence = downgrade(ctx.confidence, 'inferred');
    return `/* [${name}] */`;
  });

  // DAX IN {values} → SQL IN (values)
  sql = sql.replace(/\bIN\s+\{([^}]+)\}/gi, (match, values) => {
    const sqlValues = values.split(',').map(v => v.trim().replace(/"/g, "'")).join(', ');
    return `IN (${sqlValues})`;
  });

  // Replace DAX double-quote strings with single quotes
  sql = sql.replace(/"([^"]*)"/g, "'$1'");

  return sql;
}

function escapeRegex(str) {
  return str.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}
```

```js
/**
 * Generate a snake_case id from a measure name.
 */
function generateId(name) {
  return name
    .replace(/\$/g, 'dollar')

    .replace(/#/g, 'count')

 * Generate aliases from a measure name.

function generateAliases(name) {
  const aliases = new Set();
  const lower = name.toLowerCase().trim();

  // Base: lowercase name
  aliases.add(lower.replace(/[^a-z0-9\s]/g, ' ').replace(/\s+/g, ' ').trim());

  // Replace $ → dollars, % → percent, # → count
  let expanded = lower
    .replace(/\$/g, 'dollars')
    .replace(/%/g, 'percent')
    .replace(/#/g, 'count');
  aliases.add(expanded.replace(/[^a-z0-9\s]/g, ' ').replace(/\s+/g, ' ').trim());

  // Expand abbreviations
  let abbrevExpanded = expanded;
  for (const [abbr, full] of Object.entries(ABBREVIATIONS)) {
    const regex = new RegExp(`\\b${abbr.replace(/[+/]/g, '\\$&')}\\b`, 'gi');
    abbrevExpanded = abbrevExpanded.replace(regex, full);
  }
  const cleaned = abbrevExpanded.replace(/[^a-z0-9\s]/g, ' ').replace(/\s+/g, ' ').trim();
  if (cleaned !== lower) aliases.add(cleaned);

  // Also add a "pct" variant for %
  if (name.includes('%')) {
    aliases.add(lower.replace(/%/g, 'pct').replace(/[^a-z0-9\s]/g, ' ').replace(/\s+/g, ' ').trim());
  }

  // Remove the exact name match (alias shouldn't duplicate the name)
  const normalizedName = name.toLowerCase().replace(/[^a-z0-9\s]/g, ' ').replace(/\s+/g, ' ').trim();
  aliases.delete(normalizedName);

  return [...aliases].filter(a => a.length > 2);
}

/**
 * Map a PBIX measure table name to a glossary section.
 */
function generateSection(tableName) {
  return SECTION_MAP[tableName] || 'Other';
}

/**
 * Generate a human-readable definition from the measure name and SQL.
 */
function generateDefinition(name, sql) {
  // Extract key info from the SQL
  const tables = [];
  const sqlUpper = sql.toUpperCase();

  if (sqlUpper.includes('SUM(')) tables.push('sum');
  if (sqlUpper.includes('COUNT(')) tables.push('count');
  if (sqlUpper.includes('AVG(')) tables.push('average');
  if (sqlUpper.includes('MAX(')) tables.push('max');
  if (sqlUpper.includes('MIN(')) tables.push('min');
  if (sqlUpper.includes('NULLIF')) tables.push('ratio');
  if (sqlUpper.includes('CASE')) tables.push('conditional');
  if (sqlUpper.includes('RANK()')) tables.push('ranking');

  const aggType = tables.length > 0 ? tables.join('/') : 'calculated';

  // Build description from name
  const readable = name
    .replace(/\$/g, 'dollar amount')
    .replace(/%/g, 'percentage')
    .replace(/#/g, 'count');

  return `${readable} — ${aggType} metric`;
}

module.exports = { generateId, generateAliases, generateSection, generateDefinition };
```

```js
// scripts/translateDaxToSql.js
'use strict';

const path = require('path');
const fs = require('fs');
const { readExcel } = require('./dax-to-sql/excelReader');
const { buildDependencyGraph, topologicalSort } = require('./dax-to-sql/dependencyGraph');
const { createTranslationContext, translateMeasure } = require('./dax-to-sql/sqlTranslator');
const { generateId, generateAliases, generateSection, generateDefinition } = require('./dax-to-sql/metadataGenerator');
const { EXCLUDED_MEASURES, TABLE_MAP, TABLE_ALIAS } = require('./dax-to-sql/constants');

const EXCEL_PATH = path.resolve(__dirname, '../RTB Dataverse Technical Document 3.xlsx');
const OUTPUT_PATH = path.resolve(__dirname, '../server/context/knowledge/kpi-glossary-rtb.json');

function main() {
  console.log('=== RTB DAX-to-SQL KPI Glossary Translation ===\n');

  // 1. Parse Excel
  console.log('Step 1: Reading Excel...');
  const { measures, tableMap, columnsByTable, joinMap } = readExcel(EXCEL_PATH);
  console.log(`  Found ${measures.length} measures\n`);

  // 2. Filter out excluded measures
  const dataMeasures = measures.filter(m => !EXCLUDED_MEASURES.has(m.name));
  const excluded = measures.filter(m => EXCLUDED_MEASURES.has(m.name));
  console.log(`Step 2: Filtering...`);
  console.log(`  Excluded ${excluded.length} presentational: ${excluded.map(m => m.name).join(', ')}`);
  console.log(`  Remaining: ${dataMeasures.length} data measures\n`);

  // 3. Build dependency graph + topological sort
  console.log('Step 3: Building dependency graph...');
  const graph = buildDependencyGraph(dataMeasures);
  const { order, cycles } = topologicalSort(graph);
  console.log(`  Topological order: ${order.length} measures`);
  if (cycles.length > 0) {
    console.log(`  WARNING: ${cycles.length} circular dependencies detected: ${cycles.join(', ')}`);
  }
  console.log();

  // 4. Translate in dependency order
  console.log('Step 4: Translating DAX → T-SQL...');
  const ctx = createTranslationContext({ tableMap, tableAlias: TABLE_ALIAS, joinMap, resolvedMeasures: {} });
  const measuresByName = Object.fromEntries(dataMeasures.map(m => [m.name, m]));

  const results = {};
  const stats = { mapped: 0, inferred: 0, pbix_only: 0, warnings: [] };

  for (const name of order) {
    const measure = measuresByName[name];
    if (!measure) continue;

    const { sql, confidence, relatedTables, relatedColumns, warnings } = translateMeasure(measure.expression, ctx);

    // Store resolved SQL for inlining by dependent measures
    ctx.resolvedMeasures[name] = { sql: `(${sql})`, relatedTables, relatedColumns };

    results[name] = {
      sql,
      confidence,
      relatedTables,
      relatedColumns,
      warnings,
    };

    stats[confidence] = (stats[confidence] || 0) + 1;
    if (warnings.length > 0) {
      stats.warnings.push({ name, warnings });
    }
  }
  console.log(`  Translated: ${Object.keys(results).length} measures\n`);

  // 5. Generate JSON
  console.log('Step 5: Generating JSON...');
  const kpis = [];

  for (const name of order) {
    const measure = measuresByName[name];
    if (!measure) continue;
    const result = results[name];
    if (!result) continue;

    kpis.push({
      id: generateId(name),
      name: name,
      aliases: generateAliases(name),
      personas: ['AE', 'Manager', 'VP'],
      section: generateSection(measure.tableName),
      definition: generateDefinition(name, result.sql),
      formula: result.sql,
      components: {},
      timeVariants: [],
      relatedColumns: result.relatedColumns,
      relatedTables: result.relatedTables,
      formulaPbix: measure.expression,
      confidence: result.confidence,
      tableName: measure.tableName,
      dataType: measure.dataType,
      isHidden: measure.isHidden,
    });
  }

  // 6. Write output
  const output = JSON.stringify({ kpis }, null, 2);
  fs.writeFileSync(OUTPUT_PATH, output, 'utf-8');
  console.log(`  Written to: ${OUTPUT_PATH}`);
  console.log(`  KPIs: ${kpis.length}\n`);

  // 7. Summary report
  console.log('=== Translation Summary ===');
  console.log(`Total measures in Excel:     ${measures.length}`);
  console.log(`Excluded (presentational):   ${excluded.length}`);
  console.log(`Translated:                  ${Object.keys(results).length}`);
  console.log(`  - mapped (clean):          ${stats.mapped}`);
  console.log(`  - inferred (partial):      ${stats.inferred}`);
  console.log(`  - pbix_only (flagged):     ${stats.pbix_only || 0}`);
  console.log(`Circular dependencies:       ${cycles.length}`);
  console.log(`Measures with warnings:      ${stats.warnings.length}`);

  if (stats.warnings.length > 0) {
    console.log('\n=== Warnings ===');
    for (const { name, warnings } of stats.warnings.slice(0, 30)) {
      console.log(`  ${name}:`);
      warnings.forEach(w => console.log(`    - ${w}`));
    }
    if (stats.warnings.length > 30) {
      console.log(`  ... and ${stats.warnings.length - 30} more`);
    }
  }

  console.log('\nDone.');
}

main();
```
